        while(tela.getRollo()>0){

            tela.cortarTela(cortar);
            System.out.println("Usted corto: "+cortar+ "m" );

public class Tela{
        int rollo;

        public Tela(int rollo){
            this.rollo = rollo;
        }

        public void setRollo(int rollo){
            this.rollo = rollo; 

        }

        public int getRollo(){
            return this.rollo; 
        }

        public void cortarTela(int cantidad){
            this.setRollo(this.getRollo()-cantidad);  
        }
}

import javax.swing.*;
import java.awt.*;
import java.awt.event.*;
import java.awt.event.ActionEvent;

public class Calculadora extends JFrame implements ActionListener{
	JButton btnc1, btnc2, btnc3, btnc4;
	JLabel etiq1;
	JTextField num1, num2;
	int numero1, numero2, suma, resta, mult, div;
	//float numero;
	
	public P8(){
		setLayout(null);
		
		btnc1 = new JButton ("+");
    		btnc1.setBounds(160,50,50,30);

    		btnc2 = new JButton ("-");
    		btnc2.setBounds(230,50,50,30);

    		btnc3 = new JButton ("*");
    		btnc3.setBounds(300,50,50,30);

    		btnc4 = new JButton ("/");
    		btnc4.setBounds(370,50,50,30);
		
		etiq1 = new JLabel("0");
    		etiq1.setBounds(450,50,100,30);
		
    		num1 = new JTextField (5);
    		num1.setBounds(20,50,50,30); 

    		num2 = new JTextField (5);
    		num2.setBounds(90,50,50,30);
    	
		add(btnc1); add(btnc2); add(btnc3);add(btnc4); add(etiq1); add(num1); add(num2);
		btnc1.addActionListener(this);
		btnc2.addActionListener(this);
		btnc3.addActionListener(this);
		btnc4.addActionListener(this);
		
	}
	
	public void actionPerformed(ActionEvent e){
		numero1 = Integer.parseInt(num1.getText());
		numero2 = Integer.parseInt(num2.getText());

		if(e.getSource() == btnc1){

			suma = numero1 + numero2;
			String sumaResultado = Integer.toString(suma);
			etiq1.setText(sumaResultado);
		}
		
		if(e.getSource() == btnc2){
		
			resta = numero1 - numero2;
			String restaResultado = Integer.toString(resta);
			etiq1.setText(restaResultado);
		}

		if(e.getSource() == btnc3){
			mult = numero1 * numero2;
			String multResultado = Integer.toString(mult);
			etiq1.setText(multResultado);
		}

		if(e.getSource() == btnc4){
			div = numero1 / numero2;
			String divResultado =Integer.toString(div);
			etiq1.setText(divResultado);
		}	
	}
	
	public static void main (String args[]){
		Calculadora calc1 = new P8();
		calc1.setBounds(550,550,500,180);
		calc1.setVisible(true);

    		num2 = new JTextField (5);
